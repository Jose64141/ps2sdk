77  //Writeback data cache before loading ELF.

79  GS_BGCOLOUR = GREEN_BG;
80  ret = SifLoadElf(argv[0], &elfdata);
81  GS_BGCOLOUR = BLUE_BG;
82  if (ret == 0) {
83  GS_BGCOLOUR = YELLOW_BG;
84  SifExitRpc();
85  FlushCache(0);
86  FlushCache(2);
87 
88  GS_BGCOLOUR = PURPBLE_BG;
89  // Following the standard the first parameter of a argv is the executable itself
90  return ExecPS2((void *)elfdata.epc, (void *)elfdata.gp, argc, argv);
91  } else {
92  GS_BGCOLOUR = MAGENTA_BG;
93  SifExitRpc();
94  return -ENOENT;
95  }
